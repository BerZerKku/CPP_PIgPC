// Пост.
// Структура параметров.
    struct sStrucRpsParametr {
        char RpsType;             // тип защиты
        char RpsLine;             // тип линии
        char RpsTMan;             // допустимое время без ман
        char RpsTDelay1;          // задержка на линии1
        char RpsTImp;             // перекрытие импульсов
        char RpsPorog1;           // загрубление ДФЗ
        char RpsPeriodAmmount;    // кол-во полупериодов.

        char RpsTDelay2;          // задержка на линии2
        char RZ_WarnBarrier;      // порог предупреждения по РЗ

        char Parametr[MAX_RPS_PARAMETR - 10];
        char KS;                  // КС
        };

// Прм.
// Структура параметров.
    struct sStrucPrmParametr {
	char Delay_OnPA;	    // задержка на включение (дребезг)
        char DurationControl;       // контроль длительности
        char BlockedCom1_8Re;       // блокированные команды
        char BlockedCom9_16Re;
        char BlockedCom17_24Re;
        char BlockedCom25_32Re;
		char DelayComOff[32];  // задержка на выключение (по каждой из 32-х команд)
        char Parametr[MAX_PRM_PARAMETR - 7 - 32];
		char KS;	            // КС
        };


// Прд.
// Структура параметров
      struct sStrucPrdParametr {      //7байт
        char Delay_OnPA;	      // задержка на включение (дребезг)
		char Delay_PA;		      // длительность команды на передачу
        char BlockedCom1_8Tr;         //номера блокированных команд
        char BlockedCom9_16Tr;
        char BlockedCom17_24Tr;
        char BlockedCom25_32Tr;
        char DlitCom1_8TR;            //номера длительных команд
        char DlitCom9_16TR;
        char DlitCom17_24TR;
        char DlitCom25_32TR;
        char TestComOnOff;            // вкл/выкл тестовой команды
	char Parametr[MAX_PRD_PARAMETR - 12];
	char KS;		      // КС
      };

//  Ппп.
// Структура параметров
    struct sStrucPppParametr {
        char Desense_PRM1;            // загрубление первого ПРМ
        char IsPrmPA;                 // YES/NO вкл-выкл. удержание реле ПА
        char IsPrdPA;                 // YES/NO вкл-выкл. удержание реле ПА
        char Desense_PRM2;            // загрубление второго ПРМ
        char Delay_Restart;           // время перезапуска ПРМ (сек) ("0" - бесконечно)
        char Uout_ON_OFF;             // контроль выходного сигнала
        char CF_WarnBarrier;          // порог предупреждения по КЧ
        char DeviceNumber;            // номер аппарата
        char TimeSynch;               // синхронизация времени
		char Parametr[MAX_PPP_PARAMETR - 10];
		char KS;		      // КС
    };

	#define MAX_RPS_PARAMETR 16        // кол. байт параметров Поста
    #define MAX_PRM_PARAMETR 45        // кол. байт параметров ПРМ
    #define MAX_PRD_PARAMETR 16        // кол. байт параметров ПРД
    #define MAX_PPP_PARAMETR 16        // кол. байт параметров ППП
